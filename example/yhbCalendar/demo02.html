自定义语言： 为了不污染calendar.js文件，对语言定义将借用Jquery的extend方法来完成。 

1.新增一个语言，比如取名为abc，然后分别定义weeks等其他的属性

$.extend(calendarLang, {
    'abc':{
        'weeks':['日','壹','贰','叁','肆','伍','陸'],
        'months':['1月','2月','3月','4月','5月','6月','7月','8月','9月','10月','11月','12月'],
        'today':'今天',
        'y':'年',
        'm':'月',
        'd':'日',
        'close':'关闭',
        'spec':'假日价格',
        'soldout':'售完',
        'week_prefix':'周'
    }
});

2.在创建日期控件的时候，对语言配置为lang: "abc"
